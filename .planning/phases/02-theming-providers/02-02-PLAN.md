    Create `src/components/ThemeToggle.tsx` â€” a "use client" component that renders a glassmorphism pill button with sun/moon icon that toggles between dark and light themes.

    ```tsx
    "use client";

    import { useTheme } from "next-themes";
    import { useEffect, useState } from "react";
    import { Sun, Moon } from "lucide-react";

    export function ThemeToggle() {
      const [mounted, setMounted] = useState(false);
      const { theme, setTheme } = useTheme();

      useEffect(() => setMounted(true), []);

      if (!mounted) {
        return <div className="w-9 h-9" />;
      }

      return (
        <button
          onClick={() => setTheme(theme === "dark" ? "light" : "dark")}
          aria-label="Toggle theme"
          className="glass w-9 h-9 flex items-center justify-center rounded-full transition-colors duration-300"
        >
          {theme === "dark" ? (
            <Sun className="w-5 h-5" />
          ) : (
            <Moon className="w-5 h-5" />
          )}
        </button>
      );
    }
    ```

    Key design decisions:
    - Uses `mounted` state guard per Pitfall 16 â€” prevents hydration mismatch between SSR and client render
    - Placeholder `<div className="w-9 h-9" />` reserves exact space during SSR to prevent layout shift
    - Uses existing `.glass` class for glassmorphism pill aesthetic per CONTEXT.md decision
    - `w-9 h-9` gives 36x36px touch target (adequate for mobile)
    - `transition-colors duration-300` per CONTEXT.md 300ms transition duration decision
    - Icon swap uses conditional rendering (Sun for dark theme, Moon for light)
    - `aria-label="Toggle theme"` for accessibility

    NOTE: lucide-react may need to be installed. If the import fails, run `npm install lucide-react` before committing.
