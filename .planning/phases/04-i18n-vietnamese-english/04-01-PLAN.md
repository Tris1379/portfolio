       ```typescript
         locales: ["en", "vi"],

       });
       ```
    3. Create `src/i18n/request.ts`:
       ```typescript
       import { getRequestConfig } from "next-intl/server";
       import { routing } from "./routing";

       export default getRequestConfig(async ({ requestLocale }) => {
         let locale = await requestLocale;
         if (!locale || !routing.locales.includes(locale as "en" | "vi")) {
           locale = routing.defaultLocale;
         }
         return {
           locale,
         };
       });
       ```
       NOTE: No `messages` returned — next-intl handles routing only. Translation data comes from site.ts getConfig() in Plan 02. Components do NOT use useTranslations() — they receive resolved config.
  </action>

    ```typescript
    import { routing } from "./src/i18n/routing";

    export const config = {
      matcher: ["/((?!api|_next|_vercel|.*\\..*).*)"],
    };
    ```

    - Root `/` redirects to `/en/` (defaultLocale)

    - Matcher excludes api routes, Next.js internals, and static files

       ```typescript
         children: React.ReactNode;
         params: Promise<{ locale: string }>;

             suppressHydrationWarning
       ```
